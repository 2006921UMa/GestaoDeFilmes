# Trocar para a branch master (se necessário)

# Remover o repositório remoto antigo

git remote add origin https://github.com/2006921UMa/GestaoDeFilmes.git

# Verificar a configuração do remoto
git remote -v

# Enviar os arquivos para o GitHub na branch master
